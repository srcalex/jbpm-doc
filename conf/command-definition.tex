}

\NewEnviron{info}[1]{
	\extract{#1}%
	\begin{figure}[htp]
		\centering
		\begin{tikzpicture}
		\node[box] (m) {%
			\begin{minipage}{0.925\textwidth}
			{\LARGE\bfseries\color{dark-color}Informationsbox}\hfill
			\ifnum\pdfstrcmp{\posone}{1}=0
				\fileicon
			\fi
			\ifnum\pdfstrcmp{\postwo}{1}=0
				\downloadicon
			\fi
			\ifnum\pdfstrcmp{\posthree}{1}=0
				\bookmarkicon
			\fi\\[1em]
			\BODY
			\end{minipage}
		};
		\draw[border] (m.south west) -- (m.north west);
		\end{tikzpicture}
	\end{figure}\\
}
